# =============================================================================
# How to test these APIs
# =============================================================================
# 1. Start Auth emulator (one terminal):
#      firebase emulators:start --only auth
# 2. Start API with emulator (second terminal):
#      $env:FIREBASE_AUTH_EMULATOR_HOST = "127.0.0.1:9099"
#      go run ./cmd/api
# 3. Get a Firebase ID token and set $token (third terminal, from repo root):
#      .\scripts\get-firebase-token.ps1
#    Copy the printed "ID token" line, then in PowerShell:
#      $token = "PASTE_THE_FULL_TOKEN_HERE"
# 4. Run the commands below. All project endpoints require: Authorization = "Bearer $token"
#
# 5. Before creating projects: sync the user once so they exist in the DB (avoids
#    "projects_user_firebase_uid_fkey" error). See "Auth – sync user" below.
#
# Quick test for GET /api/v1/projects only:
#   .\scripts\get-firebase-token.ps1 -TestApi
# =============================================================================

# Notes:
# - Diagram selection: You can optionally send diagram_version_id in the POST message body.
#   If provided: uses that version for THIS message AND pins it to the thread (future messages use it too).
#   If omitted: server picks automatically:
#     - Thread with binding_mode FOLLOW_LATEST (default) => uses latest diagram.
#     - Thread with binding_mode PINNED => uses the thread's pinned diagram version.
# - Chat history is automatic: reuse the same THREAD_ID and previous turns are
#   sent to the LLM as history.
#
# - Design input: When posting the FIRST message (no chat history), you can
#   optionally include "design" with the structure:
#   { "preferred_vcpu": 4, "preferred_memory_gb": 8, "workload": { "concurrent_users": 1000 }, "budget": 2000 }
#   - If design is provided: The server will build a summary and inject it into the LLM context.
#   - If design is empty/missing: The server will add a note "Note: No design available."
#   Example with design: -Body '{"message":"...","design":{"preferred_vcpu":4,"preferred_memory_gb":8,"workload":{"concurrent_users":1000},"budget":2000}}'
#   Example without design: -Body '{"message":"..."}' (note will be added automatically)
#
# - Mode and detail (UIGP): You can send "mode" and "detail" so the user can select behavior:
#   mode: "instant" | "default" | "thinking"
#     - instant: fast, minimal reasoning
#     - default: standard behavior
#     - thinking: deeper analysis (used automatically when diagram context is sent and mode is default/omitted)
#   detail: optional, e.g. "high" for more detailed answers (defaults to "high" when diagram is sent and omitted)
#   Example: -Body '{"message":"...","mode":"thinking","detail":"high"}'
# -----------------------------------------------------------------------------

# Create chat thread (default FOLLOW_LATEST = always use latest diagram for chat)
Invoke-RestMethod -Method Post `
  -Uri "http://localhost:8000/api/v1/projects/PROJECT_PUBLIC_ID/chats" `
  -Headers @{ "Authorization" = "Bearer $token"; "Content-Type" = "application/json" } `
  -Body '{"title":"Main chat","binding_mode":"FOLLOW_LATEST"}'

# (Optional) Pin thread to a diagram version so chat uses that version instead of latest:
# - To pin the LATEST diagram: send only binding_mode (omit diagram_version_id).
# - To pin a SPECIFIC version: send diagram_version_id with the version id (e.g. dver-123).
Invoke-RestMethod -Method Patch `
  -Uri "http://localhost:8000/api/v1/projects/PROJECT_PUBLIC_ID/chats/THREAD_ID" `
  -Headers @{ "Authorization" = "Bearer $token"; "Content-Type" = "application/json" } `
  -Body '{ "binding_mode": "PINNED" }'
# Or with a specific version: -Body '{ "binding_mode": "PINNED", "diagram_version_id": "dver-xxxx" }'
